NAME Modulo8V2

VAR
  x    : boolean;
  
y    : boolean;
  
z    : boolean;


INIT
  
!x & !y & !z;



RULES
  
!z :
	
  z := true

  

z & !y:
	
  z := false; y := true

  

z & y & x:
	
  z := false; y := false; x := false

  

z & y & !x:
	
  z := false; y := false; x := true
